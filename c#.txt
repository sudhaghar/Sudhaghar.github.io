using System;
					
public class Program
{
	int num;
	/*void GetNumber()
	{
		num=Convert.ToInt32(Console.ReadLine());
	}*/ 
	void CalculateAverage()
	{
		int total=0,count=0;
		float avg;
		do
		{
			num=Convert.ToInt32(Console.ReadLine());
			if(num!=-1)
			{
				if(num%7==0)
				{
					total+=num;
					count++;
				}	
			}	
		}while(num>=0);
		avg=total/count;
		Console.WriteLine("Avg is "+avg);
	} 
	public static void Main()
	{
			Program program = new Program();
			program.CalculateAverage();
	}
}
